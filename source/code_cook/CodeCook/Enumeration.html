def permutation(array=[], i=0)
    if i == array.size
        p array
    else
        (i..array.size-1).each do |k|
            array[i], array[k] = array[k], array[i]
            permutation(array, i + 1)
            array[i], array[k] = array[k], array[i]
        end
    end
end

def subset3(array)
    n = array.size
    (1..(1 << n)).each do |i|
        (0..n).each do |j|
            print "#{array[j]} " if (i & (1 << j)) != 0
        end
        puts ''
    end
end
