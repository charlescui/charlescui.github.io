def gcd1(a, b)
    while b != 0
        c = b
        b = a % b
        a = c
    end
    a
end

def gcd2(a, b)
    while a != b
        if a > b
            a -= b
        else
            b -= a
        end
    end
    a
end

def is_prime(n)
    return 0 if n < 2
    i = 2
    while i*i <= n
        return 0 if n % i == 0
        i += 1
    end
    return 1
end
